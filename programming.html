            using namespace std;
             
            unordered_map memo;
            unordered_map> graph;
             
            // dfs(start) # => start から出発して閉路があるなら `true` を返す
            bool dfs(string s, unordered_map& visited) {
              memo[s] = true;
              if (visited[s]) return true;
              bool closed = false;
              visited[s] = true;
              for (string t : graph[s]) {
                if (dfs(t, visited)) closed = true;
              }
              visited[s] = false;
              return closed;
            }
             
            int main() {
              int n;
              cin >> n;
              vector s(n), t(n);
              for (int i = 0; i < n; i++) {
                cin >> s[i] >> t[i];
                graph[s[i]].push_back(t[i]);
              }
             
              unordered_map visited;
              for (int i = 0; i < n; i++) {
                if (memo[s[i]]) continue;
                string start = s[i];
                bool closed = dfs(start, visited);
                if (closed) {
                  cout << "No" << endl;
                  return 0;
                }
              }
              for (int i = 0; i < n; i++) {
                if (memo[t[i]]) continue;
                string start = t[i];
                bool closed = dfs(start, visited);
                if (closed) {
                  cout << "No" << endl;
                  return 0;
                }
              }
             
              cout << "Yes" << endl;
            }
